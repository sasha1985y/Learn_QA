mport requests
import pytest

BASE_URL = "https://dog.ceo/api"

▎Тест 1: Получить случайное изображение собаки

def test_get_random_dog():
    response = requests.get(f"{BASE_URL}/breeds/image/random")
    assert response.status_code == 200, f"Ожидался 200, получен {response.status_code}"
    data = response.json()
    assert "message" in data, "Нет ссылки на изображение"
    assert data["status"] == "success", "Статус не 'success'"

▎Тест 2: Получить список всех пород

def test_get_all_breeds():
    response = requests.get(f"{BASE_URL}/breeds/list/all")
    assert response.status_code == 200
    data = response.json()
    assert "message" in data, "Нет списка пород"
    assert "terrier" in data["message"], "Нет породы 'terrier'"

▎Тест 3: Получить изображения для конкретной породы (например, 'hound')

def test_get_breed_images():
    response = requests.get(f"{BASE_URL}/breed/hound/images")
    assert response.status_code == 200
    data = response.json()
    assert isinstance(data["message"], list), "Сообщение не является списком"
    assert len(data["message"]) > 0, "Нет изображений для породы"

▎Тест 4: Проверка ошибки 404 (несуществующая порода)

def test_invalid_breed():
    response = requests.get(f"{BASE_URL}/breed/invalid_breed/images")
    assert response.status_code == 404, "Ожидалась ошибка 404"

Тесты написаны на Python с использованием библиотеки requests для HTTP-запросов 
и фреймворка pytest для организации тестирования.
Код содержит четыре теста, которые проверяют разные аспекты работы API: получение

и содержания возвращаемых данных ожидаемым значениям.
